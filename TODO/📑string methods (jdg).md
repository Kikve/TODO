// Obtaining portions of a string
s.substring(1,4)
 // => "ell": the 2nd, 3rd, and 4th characters.
s.slice(1,4)
 // => "ell": same thing
s.slice(-3)
 // => "rld": last 3 characters
s.split(", ")
 // => ["Hello", "world"]: split at delimiter string
